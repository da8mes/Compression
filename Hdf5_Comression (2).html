def timer(test_code, setup, iter_num):
    total_access_time = timeit.Timer(stmt=test_code, 
                                  setup=setup).timeit(number=iter_num)
    return (total_access_time/iter_num)

def sum_four(lst):
    temp = 0
    summ = []
    for i, res in enumerate(lst):
        if (i % 4) == 3:
            temp += res[3]
            summ.append(temp)
            temp = 0
        else:
            temp += res[3]
    return summ
